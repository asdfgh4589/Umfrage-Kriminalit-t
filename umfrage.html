<h2>Von 100 Ausländer:innen: Wie viele wurden innerhalb eines Jahres einer Straftat beschuldigt?</h2>

<div class="people" id="people-foreign"></div>

<div class="slider-container">

  <input type="range" id="slider-foreign" min="0" max="100" value="0" oninput="updateSelection('foreign')" aria-label="Schätzung für Ausländer:innen">

  <span id="value-foreign" class="value">0</span> Personen

</div>

<div id="feedback-foreign" class="feedback"></div>


<button onclick="showSolution()">Lösung anzeigen</button>


<script>

  const solution = { ch: 0.7, foreign: 2.1 };


  function createPeople(containerId) {

    const container = document.getElementById(containerId);

    for (let i = 0; i < 100; i++) {

      const div = document.createElement("div");

      div.classList.add("person");

      div.setAttribute("title", "Person " + (i+1));

      container.appendChild(div);

    }

  }

  createPeople("people-ch");

  createPeople("people-foreign");


  function updateSelection(group) {

    const slider = document.getElementById("slider-" + group);

    const value = document.getElementById("value-" + group);

    const container = document.getElementById("people-" + group);

    const persons = container.getElementsByClassName("person");


    const guessed = parseFloat(slider.value);

    value.textContent = guessed;


    for (let i = 0; i < 100; i++) {

      persons[i].classList.remove("active", "solution", "half");

      if (i < guessed) {

        persons[i].classList.add("active");

      }

    }


    document.getElementById("feedback-" + group).textContent = "";

  }


  function showSolution() {

    ["ch", "foreign"].forEach(group => {

      const container = document.getElementById("people-" + group);

      const persons = container.getElementsByClassName("person");

      const correct = solution[group];


      // Feedback zur eigenen Schätzung

      const guessed = parseFloat(document.getElementById("slider-" + group).value);

      let feedback = "";

      if (guessed > 0) {

        const factor = guessed / correct;

        if (factor > 1.1) {

          feedback = `Deine Schätzung war etwa ${factor.toFixed(1)}-mal so hoch wie die Realität.`;

        } else if (factor < 0.9) {

          feedback = `Deine Schätzung lag deutlich unter der Realität (ca. ${factor.toFixed(1)}-mal zu niedrig).`;

        } else {

          feedback = `Sehr nah an der Realität geschätzt!`;

        }

      }

      document.getElementById("feedback-" + group).textContent = feedback;


      // Lösung darstellen

      const whole = Math.floor(correct);

      const hasHalf = correct % 1 >= 0.5;


      for (let i = 0; i < 100; i++) {

        persons[i].classList.remove("active", "solution", "half");

      }


      for (let i = 0; i < whole; i++) {

        persons[i].classList.add("solution");

      }


      if (hasHalf) {

        persons[whole].classList.add("half");

      }


      // Slider & Anzeige auf exakte Zahl setzen

      document.getElementById("value-" + group).textContent = correct.toFixed(1);
